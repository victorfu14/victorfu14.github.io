\begin{definition}[Differential privacy]
    A randomized algorithm $\mathcal{A}$ is $\varepsilon$-differetially private if given two adjacent data sets $B, B'$, for any subset $S \subseteq \im \mathcal{A}$, \[
        \matP[\mathcal{A}(B) \in S] \leq e^\varepsilon \cdot \matP[\mathcal{A}(B') \in S].
    \]
\end{definition}
